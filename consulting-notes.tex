%========================================================================%
% Two commands, each take a number of times to repeat.  Bullet line for  %
% optimization notes.  Task line for todos. Both lines are filled with a %
% light gray dotted underline.                                           %
%========================================================================%

\newcommand{\bulletline}[1]{
    \newcount\foo
    \foo=#1
    \loop
    \par$\bullet$ \color{lightgray} \dotuline{\hfill} \color {black}
    \advance \foo -1
    \ifnum \foo>0
    \repeat
}

\newcommand{\taskline}[1]{
    \newcount\foo
    \foo=#1
    \loop
    \par\ding{111} \color{lightgray} \dotuline{\hfill} \color{black}
    \advance \foo -1
    \ifnum \foo>0
    \repeat
}
